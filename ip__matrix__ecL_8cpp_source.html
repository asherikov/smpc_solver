00056             const double cosA,
00057             const double *i2Q,
00058             const double* i2hess)
00059     {
00060         /*      R        *       Q        *       R'      =      M
00061          * |c    -s    |   |a1          |   |c     s    |   |a1cc+a2ss     a1cs-a2cs    |
00062          * |  1        |   |   b        |   |  1        |   |          b                |
00063          * |    1      |   |     g      |   |    1      |   |            g              |
00064          * |s     c    |   |      a2    |   |-s    c    |   |a1cs-a2cs     a1ss+a2cc    |
00065          * |        1  |   |         b  |   |        1  |   |                        b  |
00066          * |          1|   |           g|   |          1|   |                          g|
00067          */
00068         // diagonal elements
00069         M[0] = i2hess[0]*cosA*cosA + i2hess[1]*sinA*sinA;
00070         /*M[28] =*/ M[7] = i2Q[1];
00071         /*M[35] =*/ M[14] = i2Q[2];
00072         M[21] = i2hess[0]*sinA*sinA + i2hess[1]*cosA*cosA;
00073 
00074         // symmetric nondiagonal elements
00075         /*M[18] =*/ M[3] = (i2hess[0] - i2hess[1])*cosA*sinA;
